package qub;

/**
 * A segment within an XML document.
 */
public interface XMLSegment
{
    default String toString(XMLFormat format)
    {
        return XML.toString(format, this::toString);
    }

    /**
     * Write the string representation of this XMLSegment to the provided stream using the provided
     * format.
     * @param stream The stream to write the string representation of this XMLSegment to.
     * @param format The format to use while writing the string.
     * @return The number of characters that were written.
     */
    Result<Integer> toString(IndentedCharacterWriteStream stream, XMLFormat format);
}
