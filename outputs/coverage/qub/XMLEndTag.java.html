package qub;

public class XMLEndTag
{
    private final String name;

    private XMLEndTag(String name)
    {
        PreCondition.assertNotNullAndNotEmpty(name, "name");

        this.name = name;
    }

    public static XMLEndTag create(String name)
    {
        PreCondition.assertNotNullAndNotEmpty(name, "name");

        return new XMLEndTag(name);
    }

    public String getName()
    {
        return this.name;
    }

    @Override
    public String toString()
    {
        return XML.toString(this::toString);
    }

    public Result<Integer> toString(IndentedCharacterWriteStream stream, XMLFormat format)
    {
        PreCondition.assertNotNull(stream, "stream");
        PreCondition.assertNotDisposed(stream, "stream");
        PreCondition.assertNotNull(format, "format");

        return Result.create(() ->
        {
            int result = 0;

            result += stream.write("</").await();
            result += stream.write(this.name).await();
            result += stream.write('>').await();

            return result;
        });
    }

    @Override
    public boolean equals(Object rhs)
    {
        return rhs instanceof XMLEndTag && this.equals((XMLEndTag)rhs);
    }

    public boolean equals(XMLEndTag rhs)
    {
        return rhs != null &&
            this.name.equals(rhs.name);
    }
}
