package qub;

public class XMLStartTag implements XMLAttributeTag, XMLAttributeContainer
{
    private final String name;
    private final XMLAttributes attributes;

    private XMLStartTag(String name)
    {
        PreCondition.assertNotNullAndNotEmpty(name, "name");

        this.name = name;
        this.attributes = XMLAttributes.create();
    }

    public static XMLStartTag create(String name)
    {
        PreCondition.assertNotNullAndNotEmpty(name, "name");

        return new XMLStartTag(name);
    }

    public static XMLStartTag create(String name, Iterable<XMLAttribute> attributes)
    {
        return XMLStartTag.create(name).setAttributes(attributes);
    }

    @Override
    public String getName()
    {
        return this.name;
    }

    @Override
    public XMLStartTag setAttribute(String attributeName, String attributeValue)
    {
        PreCondition.assertNotNullAndNotEmpty(attributeName, "attributeName");
        PreCondition.assertNotNull(attributeValue, "attributeValue");

        this.attributes.setAttribute(attributeName, attributeValue);
        return this;
    }

    @Override
    public XMLStartTag setAttribute(XMLAttribute attribute)
    {
        return (XMLStartTag)XMLAttributeTag.super.setAttribute(attribute);
    }

    @Override
    public XMLStartTag setAttributes(Iterable<XMLAttribute> attribute)
    {
        return (XMLStartTag)XMLAttributeTag.super.setAttributes(attribute);
    }

    @Override
    public Iterable<XMLAttribute> getAttributes()
    {
        return this.attributes;
    }

    @Override
    public Result<String> getAttributeValue(String attributeName)
    {
        return this.attributes.getAttributeValue(attributeName);
    }

    @Override
    public String toString()
    {
        return XML.toString(this);
    }

    @Override
    public Result<Integer> toString(IndentedCharacterWriteStream stream, XMLFormat format)
    {
        PreCondition.assertNotNull(stream, "stream");
        PreCondition.assertNotDisposed(stream, "stream");
        PreCondition.assertNotNull(format, "format");

        return Result.create(() ->
        {
            int result = 0;

            result += stream.write('<').await();
            result += stream.write(this.name).await();
            if (this.attributes.any())
            {
                result += stream.write(' ').await();
                result += this.attributes.toString(stream, format).await();
            }
            result += stream.write('>').await();

            return result;
        });
    }

    @Override
    public boolean equals(Object rhs)
    {
        return rhs instanceof XMLStartTag && this.equals((XMLStartTag)rhs);
    }

    public boolean equals(XMLStartTag rhs)
    {
        return rhs != null &&
            this.name.equals(rhs.name) &&
            this.attributes.equals(rhs.attributes);
    }
}
